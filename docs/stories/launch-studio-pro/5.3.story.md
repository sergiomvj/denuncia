### Padrﺣ۲o de agente

```ts
// lib/launch-studio-pro/agents/avatar-agent.ts
export async function runAvatarAgent(input: AvatarAgentInput, runId: string): Promise<AvatarOutput> {
  const { systemPrompt, userPrompt } = buildPrompt('avatar_agent', input);
  const result = await ModelRouter.call({
    phaseRole: 'avatar_researcher',
    systemPrompt,
    userPrompt,
    projectId: input.projectId,
    runId,
  });
  const parsed = JSON.parse(result.content);
  const validation = validateAvatarSection(parsed);
  if (!validation.valid) throw new ValidationError(validation.errors);
  return parsed as AvatarOutput;
}
```
